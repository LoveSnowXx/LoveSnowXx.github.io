#### 滚动加载

```javascript
const innerHeight = document.documentElement.scrollHeight;
    const outerHeight = document.documentElement.clientHeight;
    function getScroll(){
        return {
            top: window.pageYOffset || document.documentElement.scrollTop || document.body.scrollTop
        };
    };
    const scrollTop = getScroll().top;
    if (outerHeight + scrollTop >= innerHeight) { // 该加载数据了
    if (this.flogF) {
        this.flogF = false;
        this.page += 1;
        this.fetchGoodsList(this.catId, this.nav_id);
    }
}
```
